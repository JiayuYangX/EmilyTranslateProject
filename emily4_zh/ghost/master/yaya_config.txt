//******************************************************************************
//SHIORI/3.0设置脚本 for YAYA
//******************************************************************************

//******************************************************************************
//AI对话相关
//******************************************************************************

//------------------------------------------------------------------------------
//TALK_INTERVAL　AI对话间隔初始值[秒]
//------------------------------------------------------------------------------
#globaldefine	TALK_INTERVAL	120


//******************************************************************************
//已安装人格列表构建相关
//******************************************************************************

//------------------------------------------------------------------------------
//IGLIST_ACCEL
//	设为1则在SSP/CROW时通过NOTIFY installedghostname
//	来构建已安装人格列表。
//	不会创建sakura/kero名列表但速度快
//	设为0则通过文件扫描构建。
//	已安装人格较多时较耗时。
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.IGLIST_ACCEL	1

//------------------------------------------------------------------------------
//IGLIST_MAX
//	用文件扫描构建人格列表时，指定获取
//	数量的上限。-1为无限制（获取全部）。
//	设为0则不构建。
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.IGLIST_MAX	0


//******************************************************************************
//请求解析/响应构建相关
//******************************************************************************

//------------------------------------------------------------------------------
//SHIORI3FW.REQUEST_LINES_LIMIT
//	请求字符串解析的最大行数。
//	减少可加快启动速度，但可能无法完整获取
//	已安装的人格列表。
//	设为0则获取所有行。
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.REQUEST_LINES_LIMIT	0


//------------------------------------------------------------------------------
//SHIORI3FW.RES_REF_MAX
//	res_reference的上限
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.RES_REF_MAX			32

//------------------------------------------------------------------------------
//SHIORI3FW.AUTO_DATA_CONVERT
//	设为1则与AYA5同等处理。
//	会对SAORI返回值等进行自动类型转换、自动将\0x01转换为逗号等
//	处理。
//	设为0则不执行这些处理。
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.AUTO_DATA_CONVERT		0


//------------------------------------------------------------------------------
//SHIORI3FW.REF_ACCEL
//	设为1则不创建reference0等变量，而以reference[0]形式
//	访问。可期待略有提速。
//	设为0则创建reference0等变量（旧版兼容处理）
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.REF_ACCEL				0
